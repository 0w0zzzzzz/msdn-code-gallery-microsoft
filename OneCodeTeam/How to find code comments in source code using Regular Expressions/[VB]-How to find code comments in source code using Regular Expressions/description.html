        Dim mLine As String = String.Empty

        ' Read the file stream

        Dim reader As New StreamReader(stream)

        While reader.Peek() <> -1
            ' Read a line of the stream reader
            sLine = reader.ReadLine()

            ' Trim the space in the start of the line
